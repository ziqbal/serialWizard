** Read first analog input and send value through serial connection
** Zafar Iqbal , http://zaf.io/

// Declare and initialise variables
int y = 0;

void setup() {

    // Set digital pin 13 for output
    pinMode( 13 , OUTPUT );

    // Initialise serial port
    Serial.begin( 9600 );

}

void loop() {

    // Read data from analog 0
    y = analogRead( 0 );

    // Send data
    Serial.println( y );

    // Blink and take a breath
    digitalWrite( 13 , HIGH );
    delay( 10 );
    digitalWrite( 13 , LOW );
    delay( 10 );

}
